52  }
53 
54  // verify CRC
55  uint16_t crc = crc16Data(_proto.buf, _proto.dataLen-2, 0);
56  uint16_t resp_crc = *(uint16_t*)(_proto.buf+_proto.dataLen-2);
57  if (crc != resp_crc) {
58  DBG("ELC: Invalid CRC");
59  return NULL;
60  }
61 
62  // dispatch based on command
63  if (packet->cmd == CMD_RESP_V) {
64  // value response
65  if (_debugEn) {
66  _debug->print("RESP_V: ");
67  _debug->println(packet->value);
68  }
69  return packet;
70  } else if (packet->cmd == CMD_RESP_CB) {
71  FP<void, void*> *fp;
72  // callback reponse
73  if (_debugEn) {
74  _debug->print("RESP_CB: ");
75  _debug->print(packet->value);
76  _debug->print(" ");
77  _debug->println(packet->argc);
78  }
79  fp = (FP<void, void*>*)packet->value;
80  if (fp->attached()) {
81  ELClientResponse resp(packet);
82  (*fp)(&resp);
83  }
84  return NULL;
85  } else {
86  if( callbackPacketHandler != NULL )
87  {
88  if( callbackPacketHandler(packet) )
89  return NULL;
90  }
91  // command (NOT IMPLEMENTED)
92  if (_debugEn) _debug->println("CMD??");
93  return NULL;
94  }
95 }

117 ELClientPacket *ELClient::Process() {
118  int value;
119  while (_serial->available()) {
120  value = _serial->read();
121  if (value == SLIP_ESC) {
122  _proto.isEsc = 1;
123  } else if (value == SLIP_END) {
124  ELClientPacket *packet = _proto.dataLen >= 8 ? protoCompletedCb() : 0;
125  _proto.dataLen = 0;
126  _proto.isEsc = 0;
127  if (packet != NULL) return packet;
128  } else {
129  if (_proto.isEsc) {
130  if (value == SLIP_ESC_END) value = SLIP_END;
131  if (value == SLIP_ESC_ESC) value = SLIP_ESC;
132  _proto.isEsc = 0;
133  }
134  if (_proto.dataLen < _proto.bufSize) {
135  _proto.buf[_proto.dataLen++] = value;
136  }
137  }
138  }
139  return NULL;
140 }

491 uint16_t ELClient::crc16Add(unsigned char b, uint16_t acc)
492 {
493  acc ^= b;
494  acc = (acc >> 8) | (acc << 8);
495  acc ^= (acc & 0xff00) << 4;
496  acc ^= (acc >> 8) >> 4;
497  acc ^= (acc & 0xff00) >> 5;
498  return acc;
499 }
