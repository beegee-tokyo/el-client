249 void ELClientWebServer::processPacket(ELClientPacket *packet)
250 {
251  ELClientResponse response(packet);
252 
253  uint16_t shrt;
254  response.popArg(&shrt, 2);
255  RequestReason reason = (RequestReason)shrt; // request reason
256 
257  response.popArg(remote_ip, 4); // remote IP address
258  response.popArg(&remote_port, 2); // remote port
259 
260  char * url;
261  int urlLen = response.popArgPtr(&url);
262 
263  WebServerCallback callback = 0;
264 
265  struct _Handler *hnd = handlers;
266  while( hnd != 0 )
267  {
268  if( hnd->URL.length() == urlLen && memcmp( url, hnd->URL.begin(), urlLen ) == 0 )
269  {
270  callback = hnd->callback;
271  break;
272  }
273  hnd = hnd->next;
274  }
275 
276  if( hnd == 0 ) // no handler found for the URL
277  {
278  _elc->_debug->print(F("Handler not found for URL:"));
279 
280  for(int i=0; i < urlLen; i++)
281  _elc->_debug->print( url[i] );
282  _elc->_debug->println();
283  return;
284  }
285 
286  switch(reason)
287  {
288  case WS_BUTTON: // invoked when a button pressed
289  {
290  char * idPtr;
291  int idLen = response.popArgPtr(&idPtr);
292 
293  char id[idLen+1];
294  memcpy(id, idPtr, idLen);
295  id[idLen] = 0;
296 
297  callback(BUTTON_PRESS, id, idLen);
298  }
299  break;
300  case WS_SUBMIT: // invoked when a form submitted
301  {
302  int cnt = 4;
303 
304  while( cnt < response.argc() )
305  {
306  char * idPtr;
307  int idLen = response.popArgPtr(&idPtr);
308  int nameLen = strlen(idPtr+1);
309  int valueLen = idLen - nameLen -2;
310 
311  arg_ptr = (char *)malloc(valueLen+1);
312  arg_ptr[valueLen] = 0;
313  memcpy(arg_ptr, idPtr + 2 + nameLen, valueLen);
314 
315  callback(SET_FIELD, idPtr+1, nameLen);
316 
317  free(arg_ptr);
318  arg_ptr = 0;
319  cnt++;
320  }
321  }
322  return;
323  case WS_LOAD: // invoked at refresh / load
324  case WS_REFRESH:
325  break;
326  default:
327  return;
328  }
329 
330  // the response is generated here with the fields to refresh
331 
332  _elc->Request(CMD_WEB_DATA, 100, 255); // 255 arguments (can't foretell, so use maximum)
333  _elc->Request(remote_ip, 4); // send remote IP address
334  _elc->Request((uint8_t *)&remote_port, 2); // send remote port
335 
336  callback( reason == WS_LOAD ? LOAD : REFRESH, NULL, 0); // send arguments
337 
338  _elc->Request((uint8_t *)NULL, 0); // end indicator
339  _elc->Request(); // finish packet
340 }
